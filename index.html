Hàm gọi dữ liệu dùng chung (Vượt rào cản CORS) async function callGoogle(params) { const url = API_URL + "?" + new URLSearchParams(params); // Dùng mode: 'no-cors' hoặc fetch thông thường qua URLSearchParams const res = await fetch(url); return await res.json(); } // 3. Hàm tra cứu tên nhân viên async function lookupEmployee() { const id = document.getElementById('user-nvId').value.trim(); if(id.length >= 3) { const name = await callGoogle({ action: 'getName', id: id }); currentEmployeeName = name; const d = document.getElementById('display-nv-name'); d.innerText = "👤 " + name; d.style.color = name === "Vô danh tiểu tốt" ? "#e53e3e" : "#38a169"; } } // 4. Hàm gửi báo cáo Nhận/Trả async function userSubmit(action) { const nvId = document.getElementById('user-nvId').value.trim(); const tbId = document.getElementById('user-tbId').value.trim(); if(!nvId || !tbId || !base64Image) return alert("Thiếu thông tin hoặc ảnh!"); const data = { nvId, nvName: currentEmployeeName, tbId, tinhTrang: document.getElementById('user-tinhTrang').value, hanhDong: action, hinhAnh: base64Image }; // Gửi qua phương thức GET (để tránh lỗi POST trên GitHub) const res = await callGoogle({ action: 'submit', data: JSON.stringify(data) }); alert(res); location.reload(); }
